SMCController.m 
function [u, s] = compute(obj, e, e_dot)
    % Sliding surface: s = e_dot + lambda * e
    s = e_dot + obj.lambda * e;
    
    % Equivalent Control
    u_eq = obj.lambda * e_dot;
    
    % Switching Control (Saturation for Chattering)
    if abs(s) <= obj.phi
        sat_s = s / obj.phi;
    else
        sat_s = sign(s);
    end
    
    u_sw = obj.eta * sat_s;
    u = u_eq + u_sw;
end
